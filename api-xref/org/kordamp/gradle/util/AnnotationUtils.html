74      @Nonnull
75      public static List<Annotation> harvestQualifiers(@Nonnull Class<?> klass) {
76          requireNonNull(klass, ERROR_CLASS_NULL);
77          List<Annotation> list = new ArrayList<>();
78          Annotation[] annotations = klass.getAnnotations();
79          for (Annotation annotation : annotations) {
80              if (AnnotationUtils.isAnnotatedWith(annotation, Qualifier.class)) {
81                  // special case for @Named
82                  if (Named.class.isAssignableFrom(annotation.getClass())) {
83                      Named named = (Named) annotation;
84                      if (isBlank(named.value())) {
85                          list.add(named(getPropertyName(klass)));
86                          continue;
87                      }
88                  }
89                  list.add(annotation);
90              }
91          }
92          return list;
93      }

100     @Nullable
101     public static <A extends Annotation> A findAnnotation(@Nonnull Class<?> klass, @Nonnull Class<A> annotationType, boolean deep) {
102         requireNonNull(klass, ERROR_CLASS_NULL);
103         requireNonNull(annotationType, ERROR_ANNOTATION_TYPE_NULL);
104 
105         while (klass != null) {
106             Annotation annotation = findAnnotation(klass.getAnnotations(), annotationType, deep);
107             if (annotation != null) {
108                 return (A) annotation;
109             }
110             klass = klass.getSuperclass();
111         }
112         return null;
113     }

120     @Nullable
121     public static <A extends Annotation> A findAnnotation(@Nonnull Method method, @Nonnull Class<A> annotationType, boolean deep) {
122         requireNonNull(method, ERROR_METHOD_NULL);
123         requireNonNull(annotationType, ERROR_ANNOTATION_TYPE_NULL);
124 
125         for (Annotation annotation : method.getAnnotations()) {
126             if (annotationType.equals(annotation.annotationType())) {
127                 return (A) annotation;
128             }
129             if (deep) {
130                 A a = findAnnotation(annotation.annotationType().getAnnotations(), annotationType, deep);
131                 if (a != null) {
132                     return a;
133                 }
134             }
135         }
136 
137         return null;
138     }

195         requireNonNull(annotation, ERROR_ANNOTATION_NULL);
196         requireNonNull(annotationType, ERROR_ANNOTATION_TYPE_NULL);
197 
198         for (Annotation a : annotation.annotationType().getAnnotations()) {
199             if (annotationType.equals(a.annotationType())) {
200                 return true;
201             }
202

204                 if (isAnnotatedWith(a, annotationType)) {
205                     return true;
206                 }
207             }
208         }
209         return false;

212     public static boolean isAnnotatedWith(@Nonnull Class<?> clazz, @Nonnull Class<? extends Annotation> annotationType) {
213         requireNonNull(clazz, ERROR_CLASS_NULL);
214         requireNonNull(annotationType, ERROR_ANNOTATION_TYPE_NULL);
215 
216         while (clazz != null) {
217             for (Annotation annotation : clazz.getAnnotations()) {
218                 if (annotationType.equals(annotation.annotationType())) {
219                     return true;
220                 }
221             }
222             for (Class<?> iface : clazz.getInterfaces()) {
223                 if (isAnnotatedWith(iface, annotationType)) {
224                     return true;
225                 }
226             }
227 
228             clazz = clazz.getSuperclass();
229         }
230         return false;
231     }

346     @Nonnull
347     public static String[] namesFor(@Nonnull Method setterMethod) {
348         requireNonNull(setterMethod, ERROR_SETTER_METHOD_NULL);
349 
350         Class<?>[] parameterTypes = setterMethod.getParameterTypes();
351         requireState(parameterTypes != null && parameterTypes.length > 0, "Argument 'setterMethod' must have at least one parameter. " + MethodDescriptor.forMethod(setterMethod));
352 
353 
354         List<String> names = new ArrayList<>();
355         Named annotation = findAnnotation(annotationsOfMethodParameter(setterMethod, 0), Named.class);
356         if (annotation != null && isNotBlank(annotation.value())) {
357             names.add(annotation.value());
358         } else {
359             if (isSetterMethod(setterMethod)) {
360                 names.add(uncapitalize(setterMethod.getName().substring(3)));
361             } else {
362                 names.add(uncapitalize(setterMethod.getName()));
363             }
364         }
365         names.add(parameterTypes[0].getName());
366         return names.toArray(new String[names.size()]);
367     }

387         Map<String, T> map = new LinkedHashMap<>();

396     @Nonnull
397     public static <T> Map<String, T> mapInstancesByName(@Nonnull String context, @Nonnull Collection<T> instances, @Nonnull String suffix, @Nonnull String type) {
398         Map<String, T> map = new LinkedHashMap<>();
399 
400         for (T instance : instances) {
401             String currentEvicts = getEvicts(instance);
402             String name = getLogicalPropertyName(nameFor(instance), suffix);
403             Object evictedInstance = isBlank(currentEvicts) ? null : map.get(currentEvicts);
404 
405             if (evictedInstance != null) {
406                 String evictedEvicts = getEvicts(evictedInstance);
407                 if (isNotBlank(evictedEvicts)) {
408                     throw new IllegalArgumentException(type + " " + name + " has an eviction conflict between " + instance + " and " + evictedInstance);
409                 } else {
410                     name = currentEvicts;
411                     LOG.info("[" + context + "] {} {} with instance {} evicted by {}", type, name, evictedInstance, instance);
412                     map.put(name, instance);
413                 }
414             } else {
415                 name = isBlank(currentEvicts) ? name : currentEvicts;
416                 Object previousInstance = map.get(name);
417                 if (previousInstance != null) {
418                     if (isBlank(getEvicts(previousInstance))) {
419                         throw new IllegalArgumentException(type + " " + name + " neither " + instance + " nor " + previousInstance + " is marked with @Evict");
420                     } else {
421                         LOG.info("[" + context + "] {} {} with instance {} evicted by {}", type, name, instance, previousInstance);
422                     }
423                 } else {
424                     map.put(name, instance);
425                 }
426             }
427         }
428 
429         return map;
430     }

437     @Nonnull
438     public static <T> Map<String, T> sortByDependencies(@Nonnull String context, @Nonnull Collection<T> instances, @Nonnull String suffix, @Nonnull String type, @Nonnull List<String> order) {
439         requireNonNull(instances, "Argument 'instances' must not be null");
440         requireNonNull(suffix, ERROR_SUFFIX_NULL);
441         requireNonNull(type, "Argument 'type' must not be null");
442         requireNonNull(order, "Argument 'order' must not be null");
443 
444         Map<String, T> instancesByName = mapInstancesByName(context, instances, suffix, type);
445 
446         Map<String, T> map = new LinkedHashMap<>(instancesByName);
447 
448         if (!order.isEmpty()) {
449             Map<String, T> tmp1 = new LinkedHashMap<>(instancesByName);
450             Map<String, T> tmp2 = new LinkedHashMap<>();
451             //noinspection ConstantConditions
452             for (String name : order) {
453                 if (tmp1.containsKey(name)) {
454                     tmp2.put(name, tmp1.remove(name));
455                 }
456             }
457             tmp2.putAll(tmp1);
458             map.clear();
459             map.putAll(tmp2);
460         }
461 
462         List<T> sorted = new ArrayList<>();
463         Set<String> instanceDeps = new LinkedHashSet<>();
464 
465         while (!map.isEmpty()) {
466             int processed = 0;
467 
468             LOG.info("[" + context + "] Current {} order is {}", type, instancesByName.keySet());
469 
470             for (Iterator<Map.Entry<String, T>> iter = map.entrySet().iterator(); iter.hasNext(); ) {
471                 Map.Entry<String, T> entry = iter.next();
472                 String instanceName = entry.getKey();
473                 String[] dependsOn = getDependsOn(entry.getValue());
474 
475                 LOG.debug("[" + context + "] Processing {} '{}'", type, instanceName);
476                 LOG.debug("[" + context + "]   depends on '{}'", Arrays.toString(dependsOn));
477 
478                 if (dependsOn.length != 0) {
479                     LOG.debug("[" + context + "]   checking {} '{}' dependencies ({})", type, instanceName, dependsOn.length);
480 
481                     boolean failedDep = false;
482                     for (String dep : dependsOn) {
483                         LOG.debug("[" + context + "]   checking {} '{}' dependency {}", type, instanceName, dep);
484                         if (!instanceDeps.contains(dep)) {
485                             // dep not in the list yet, we need to skip adding this to the list for now
486                             LOG.debug("[" + context + "]   skipped {} '{}', since dependency '{}' not yet added", type, instanceName, dep);
487                             failedDep = true;
488                             break;
489                         } else {
490                             LOG.debug("[" + context + "]   {} '{}' dependency '{}' already added", type, instanceName, dep);
491                         }
492                     }
493 
494                     if (failedDep) {
495                         // move on to next dependency
496                         continue;
497                     }
498                 }
499 
500                 LOG.debug("[" + context + "]   adding {} '{}', since all dependencies have been added", type, instanceName);
501                 sorted.add(entry.getValue());
502                 instanceDeps.add(instanceName);
503                 iter.remove();
504                 processed++;
505             }
506 
507             if (processed == 0) {
508                 // we have a cyclical dependency, warn the user and load in the order they appeared originally
509                 LOG.debug("[" + context + "]   unresolved {} dependencies detected", type);
510                 for (Map.Entry<String, T> entry : map.entrySet()) {
511                     String instanceName = entry.getKey();
512                     String[] dependsOn = getDependsOn(entry.getValue());
513 
514                     // display this as a cyclical dep
515                     LOG.debug("[" + context + "]   {} {} ", type, instanceName);
516                     if (dependsOn.length != 0) {
517                         for (String dep : dependsOn) {
518                             LOG.debug("    depends on {}", dep);
519                         }
520                     } else {
521                         // we should only have items left in the list with deps, so this should never happen
522                         // but a wise man once said...check for true, false and otherwise...just in case
523                         LOG.debug("[" + context + "]   problem while resolving dependencies.");
524                         LOG.debug("[" + context + "]   unable to resolve dependency hierarchy.");
525                     }
526                     if (!instanceDeps.contains(instanceName)) {
527                         LOG.debug("[" + context + "]   adding {} '{}' at end", type, instanceName);
528                         sorted.add(entry.getValue());
529                         instanceDeps.add(instanceName);
530                     }
531                 }
532                 break;
533                 // if we have processed all the instances, we are done
534             } else if (sorted.size() == instancesByName.size()) {
535                 LOG.info("[" + context + "] {} dependency ordering complete", type);
536                 break;
537             }
538         }
539 
540         instancesByName = mapInstancesByName(sorted, suffix);
541         LOG.info("[" + context + "] Computed {} order is {}", type, instancesByName.keySet());
542 
543         return instancesByName;
544     }

551     @SuppressWarnings("ClassExplicitlyAnnotation")
552     private static class NamedImpl implements Named, Serializable {
553         private static final long serialVersionUID = 0;
554         private final String value;
555 
556         public NamedImpl(String value) {
557             this.value = requireNonNull(value, VALUE);
558         }
559 
560         public String value() {
561             return this.value;
562         }
563 
564         public int hashCode() {
565             // This is specified in java.lang.Annotation.
566             return (127 * VALUE.hashCode()) ^ value.hashCode();
567         }
568 
569         public boolean equals(Object o) {
570             if (!(o instanceof Named)) {
571                 return false;
572             }
573 
574             Named other = (Named) o;
575             return value.equals(other.value());
576         }
577 
578         public String toString() {
579             return "@" + Named.class.getName() + "(value=" + value + ")";
580         }
581 
582         public Class<? extends Annotation> annotationType() {
583             return Named.class;
584         }
