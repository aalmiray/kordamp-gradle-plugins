047 /**
048  * Based on https://github.com/apache/maven-artifact-plugin/blob/master/src/main/java/org/apache/maven/plugins/artifact/buildinfo/BuildInfoWriter.java
049  * Original author: Herve Boutemy
050  *
051  * @author Andres Almiray
052  * @since 0.43.0
053  */
054 @CompileStatic
055 class CreateBuildInfoTask extends DefaultTask {
056     @OutputFile
057     final RegularFileProperty outputFile
058 
059     @InputFiles
060     @Optional
061     final ConfigurableFileCollection additionalArtifacts
062 
063     @Input
064     @Optional
065     final MapProperty<String, Object> additionalProperties
066 
067     @Inject
068     CreateBuildInfoTask(ObjectFactory objects) {
069         outputFile = objects.fileProperty()
070         additionalArtifacts = objects.fileCollection()
071         additionalProperties = objects.mapProperty(String, Object).convention([:])
072     }
073 
074     @TaskAction
075     void generateFile() {
076         Collection<File> files = resolveFiles()
077 
078         if (!files) {
079             return
080         }
081 
082         File output = outputFile.asFile.get()
083         output.parentFile.mkdirs()
084 
085         Writer writer = new PrintWriter(new FileOutputStream(output))
086         printHeader(writer)
087         printArtifacts(writer, files)
088         writer.flush()
089         writer.close()
090     }
091 
092     @CompileDynamic
093     private Collection<File> resolveFiles() {
094         Collection<File> files = []
095         PublishingExtension publishing = project.extensions.findByType(PublishingExtension)
096 
097         if (publishing) {
098             publishing.publications.each { Publication publication ->
099                 if (publication instanceof MavenPublication) {
100                     MavenPublication mvn = (MavenPublication) publication
101                     mvn.artifacts.each { MavenArtifact artifact ->
102                         if ('buildinfo' != artifact.extension &&
103                             'sources' != artifact.classifier) {
104                             files.add(artifact.file)
105                         }
106                     }
107                 }
108             }
109         }
110 
111         files.addAll(additionalArtifacts.files)
112 
113         files.unique().sort { File f -> f.name }
114     }
115 
116     private void printHeader(Writer writer) {
117         writer.println('# https://reproducible-builds.org/docs/jvm/')
118         writer.println('buildinfo.version=1.0-SNAPSHOT')
119         writer.println()
120         writer.println("name=${project.name}")
121         writer.println("group-id=${project.group}")
122         writer.println("artifact-id=${project.name}")
123         writer.println("version=${project.version}")
124         writer.println()
125         boolean scmUsed = printSourceInformation(writer)
126         writer.println()
127         writer.println('# build instructions')
128         writer.println('build-tool=gradle')
129         writer.println()
130         printAdditionalProperties(writer)
131         writer.println('# effective build environment information')
132         writer.println("java.version=${System.properties['java.version']}")
133         writer.println("java.vendor=${System.properties['java.vendor']}")
134         writer.println("os.name=${System.properties['os.name']}")
135         if (scmUsed) writer.println('source.used=scm')
136         writer.println()
137         printToolInformation(writer)
138     }
139 
140     private boolean printSourceInformation(Writer writer) {
141         ProjectConfigurationExtension config = resolveConfig(project)
142 
143         writer.println('# source information')
144 
145         String scmUrl = config.info.scm.connection
146         if (isBlank(scmUrl)) scmUrl = config.info.links.scm
147 
148         if (isNotBlank(scmUrl)) {
149             writer.println("source.scm.uri=${scmUrl}")
150             if (isNotBlank(config.info.scm.tag)) {
151                 writer.println("source.scm.tag=${config.info.scm.tag}")
152                 if (String.valueOf(project.version).endsWith('-SNAPSHOT')) {
153                     project.logger.warn("SCM source tag in buildinfo source.scm.tag=" + config.info.scm.tag
154                         + " does not permit rebuilders reproducible source checkout")
155                 }
156             }
157             return true
158         }
159 
160         writer.println("# no scm configured for ${project.path}")
161         project.logger.warn("No source information available in buildinfo for rebuilders...")
162 
163         false
164     }
165 
166     private void printAdditionalProperties(Writer writer) {
167         Map<String, Object> props = additionalProperties.get()
168         if (props) {
169             writer.println('# additional properties')
170             additionalProperties.get().each { key, value ->
171                 writer.println("${key}=${value}")
172             }
173             writer.println()
174         }
175         project.gradle.startParameter
176     }
177 
178     private void printToolInformation(Writer writer) {
179         writer.println('# Gradle rebuild instructions and effective environment')
180         writer.println("gradle.version=${GradleVersion.current().version} (${GradleVersion.current().revision})")
181         writer.println()
182     }
183 
184     private void printArtifacts(Writer writer, Collection<File> files) {
185         files.eachWithIndex { File file, int i ->
186             writer.println("outputs.${i}.filename=${file.name}")
187             writer.println("outputs.${i}.length=${file.length()}")
188             writer.println("outputs.${i}.checksums.md5=${calculateHash(file, 'MD5')}")
189             writer.println("outputs.${i}.checksums.sha1=${calculateHash(file, 'SHA1')}")
190             writer.println("outputs.${i}.checksums.sha256=${calculateHash(file, 'SHA-256')}")
191             writer.println("outputs.${i}.checksums.sha512=${calculateHash(file, 'SHA-512')}")
192         }
193     }
194 
195     private static calculateHash(File file, String algorithm) {
196         file.withInputStream {
197             new DigestInputStream(it, MessageDigest.getInstance(algorithm)).withStream {
198                 it.eachByte {}
199                 it.messageDigest.digest().encodeHex() as String
200             }
201         }
202     }
