047 /**
048  * @author Andres Almiray
049  * @since 0.16.0
050  */
051 @CompileStatic
052 class PluginPlugin extends AbstractKordampPlugin {
053     Project project
054 
055     PluginPlugin() {
056         super(Plugins.PLUGIN_ID)
057     }
058 
059     void apply(Project project) {
060         this.project = project
061 
062         configureProject(project)
063         project.childProjects.values().each {
064             it.pluginManager.apply(PluginPlugin)
065         }
066     }
067 
068     private void configureProject(Project project) {
069         if (hasBeenVisited(project)) {
070             return
071         }
072         setVisited(project, true)
073 
074         BasePlugin.applyIfMissing(project)
075 
076         project.tasks.register('listPluginDescriptors', ListPluginDescriptorsTask.class,
077             new Action<ListPluginDescriptorsTask>() {
078                 void execute(ListPluginDescriptorsTask t) {
079                     t.group = 'Plugin development'
080                     t.description = 'Lists plugin descriptors from plugin declarations.'
081                     t.declarations.set(project.extensions.findByType(GradlePluginDevelopmentExtension).plugins)
082                 }
083             })
084 
085         addProjectEvaluatedListener(project, new PublishingProjectEvaluatedListener())
086 
087         project.pluginManager.apply(JavaGradlePluginPlugin)
088         project.pluginManager.apply(PublishPlugin)
089 
090         // Must execute after JavaGradlePluginPlugin|PublishPlugin
091         // because ${plugin.name}PluginMarkerMaven is created explicitly and not with `maybeCreate`.
092         project.afterEvaluate {
093             ProjectConfigurationExtension config = resolveConfig(project)
094             config.plugins.plugins.values().each { plugin ->
095                 String pluginName = plugin.name
096                 updatePluginPublication(project, config, pluginName)
097             }
098             updatePublications(project, config)
099         }
100     }
101 
102     @Named('plugin')
103     //@DependsOn(['publishing'])
104     private class PublishingProjectEvaluatedListener implements ProjectEvaluatedListener {
105         @Override
106         void projectEvaluated(Project project) {
107             ProjectConfigurationExtension config = resolveConfig(project)
108 
109             GradlePluginDevelopmentExtension gpdExt = project.extensions.findByType(GradlePluginDevelopmentExtension)
110             PluginBundleExtension pbExt = project.extensions.findByType(PluginBundleExtension)
111 
112             if (isBlank(pbExt.website) && isNotBlank(config.info.url)) {
113                 pbExt.website = config.info.url
114             }
115             if (isBlank(pbExt.vcsUrl)) {
116                 pbExt.vcsUrl = config.info.resolveScmLink()
117             }
118             if (isBlank(pbExt.description)) {
119                 pbExt.description = config.info.description
120             }
121             if (!pbExt.tags) {
122                 pbExt.tags = config.info.tags
123             }
124 
125             if (isBlank(pbExt.mavenCoordinates.groupId)) {
126                 pbExt.mavenCoordinates.groupId = project.group
127             }
128             if (isBlank(pbExt.mavenCoordinates.artifactId)) {
129                 pbExt.mavenCoordinates.artifactId = project.name
130             }
131 
132             if (isBlank(pbExt.mavenCoordinates.version)) {
133                 pbExt.mavenCoordinates.version = project.version
134             }
135 
136             config.plugins.plugins.values().each { plugin ->
137                 String pluginName = plugin.name
138 
139                 PluginDeclaration pd = gpdExt.plugins.maybeCreate(pluginName)
140                 pd.id = plugin.id
141                 pd.displayName = plugin.displayName ?: config.info.description
142                 pd.description = plugin.description ?: config.info.description
143                 pd.implementationClass = plugin.implementationClass
144 
145                 PluginConfig pc = pbExt.plugins.maybeCreate(pluginName)
146                 pc.id = plugin.id
147                 pc.displayName = plugin.displayName ?: config.info.description
148                 pc.description = plugin.description ?: config.info.description
149                 pc.tags = plugin.resolveTags(config)
150                 pc.version = project.version
151             }
152         }
153     }
154 
155     @CompileDynamic
156     private void updatePluginPublication(Project project, ProjectConfigurationExtension config, String pluginName) {
157         PublishingUtils.configureSigning(config, project, pluginName + 'PluginMarkerMaven')
158     }
159 
160     @CompileDynamic
161     private void updatePublications(Project project, ProjectConfigurationExtension config) {
162         project.publishing {
163             publications {
164                 pluginMaven(MavenPublication) {
165                     PublishingUtils.configurePom(pom, config, config.publishing.pom)
166                 }
167             }
168         }
169 
170         PublishingUtils.configureSigning(config, project, 'pluginMaven')
171     }
