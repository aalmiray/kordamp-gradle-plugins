113     static void merge(Information o1, Information o2) {
114         o2.normalize()
115         o1.name = o1.@name ?: o2.name
116         o1.description = o1.description ?: o2.description
117         o1.inceptionYear = o1.@inceptionYear ?: o2.inceptionYear
118         o1.vendor = o1.@vendor ?: o2.vendor
119         o1.tags = CollectionUtils.merge(o1.tags, o2.tags, false)
120         Specification.merge(o1.spec, o2.spec)
121         Implementation.merge(o1.impl, o2.impl)
122         Organization.merge(o1.organization, o2.organization)
123         PersonSetImpl.merge(o1.@people, o2.@people)
124         RepositorySetImpl.merge(o1.@repositories, o2.@repositories)
125         Links.merge(o1.links, o2.links)
126         Scm.merge(o1.scm, o2.scm)
127         IssueManagement.merge(o1.issueManagement, o2.issueManagement)
128         CiManagement.merge(o1.ciManagement, o2.ciManagement)
129         MailingListSetImpl.merge(o1.@mailingLists, o2.@mailingLists)
130         CredentialsSetImpl.merge(o1.@credentials, o2.@credentials)
131         o1.normalize()
132     }

134     List<String> validate(ProjectConfigurationExtension extension) {
135         List<String> errors = []
136 
137         if (isBlank(description) &&
138             (extension.publishing.enabled || extension.bintray.enabled)) {
139             errors << "[${project.name}] Project description is blank".toString()
140         }
141         if (isBlank(getVendor()) &&
142             (extension.publishing.enabled || extension.bintray.enabled)) {
143             errors << "[${project.name}] Project vendor is blank".toString()
144         }
145         if (isBlank(getUrl()) &&
146             (extension.publishing.enabled || extension.bintray.enabled)) {
147             errors << "[${project.name}] Project organization.url is blank".toString()
148         }
149         if ((scm.enabled && isBlank(scm.url)) && (links.enabled && isBlank(links.scm)) &&
150             (extension.publishing.enabled || extension.bintray.enabled)) {
151             errors << "[${project.name}] Project scm.url is blank".toString()
152         }
153 
154         errors
155     }

261     Information normalize() {
262         if (spec.isEnabledSet()) specification.setEnabled(spec.enabled)
263         specification.title = spec.title ?: project.name
264         specification.version = spec.version ?: project.version
265         specification.vendor = spec.vendor ?: getVendor()
266 
267         if (impl.isEnabledSet()) implementation.setEnabled(impl.enabled)
268         implementation.title = impl.title ?: project.name
269         implementation.version = impl.version ?: project.version
270         implementation.vendor = impl.vendor ?: getVendor()
271 
272         this
273     }

275     List<String> getAuthors() {
276         List<String> authors = []
277 
278         people.forEach { Person person ->
279             if ('author' in person.roles*.toLowerCase()) {
280                 String author = person.name ?: person.id
281                 if (author) authors << author
282             }
283         }
284 
285         if (!authors && !people.isEmpty()) {
286             Person person = people.people[0]
287             authors << person.name ?: person.id
288         }
289 
290         authors
291     }

293     String getCopyrightYear() {
294         String initialYear = getInceptionYear()
295         String currentYear = currentYear()
296         String year = initialYear
297         if (initialYear != currentYear) {
298             year += '-' + currentYear
299         }
300         year
301     }

303     static String currentYear() {
304         Date now = new Date()
305         Calendar c = Calendar.getInstance()
306         c.setTime(now)
307         return c.get(Calendar.YEAR).toString()
308     }
