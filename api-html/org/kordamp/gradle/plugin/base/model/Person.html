19 
20 import groovy.transform.CompileStatic
21 import org.gradle.api.Action
22 import org.kordamp.gradle.util.CollectionUtils
23 import org.kordamp.gradle.util.ConfigureUtil
24 
25 /**
26  * @author Andres Almiray
27  * @since 0.8.0
28  */
29 @CompileStatic
30 class Person {
31     String id
32     String name
33     String email
34     String url
35     String timezone
36     Organization organization
37     List<String> roles = []
38     Map<String, String> properties = [:]
39 
40     Map<String, Object> toMap() {
41         new LinkedHashMap<String, Object>([
42             id          : id,
43             name        : name,
44             email       : email,
45             url         : url,
46             timezone    : timezone,
47             organization: organization?.toMap(),
48             roles       : roles,
49             properties  : properties
50         ])
51     }
52 
53     static Person merge(Person o1, Person o2) {
54         o1.id = o1.id ?: o2?.id
55         o1.name = o1.name ?: o2?.name
56         o1.email = o1.email ?: o2?.email
57         o1.url = o1.url ?: o2?.url
58         o1.timezone = o1.timezone ?: o2?.timezone
59         o1.roles = CollectionUtils.merge(o1.roles, o2?.roles, false)
60         if (o1.organization) {
61             Organization.merge(o1.organization, o2?.organization)
62         } else {
63             o1.organization = o2?.organization
64         }
65         o1.properties = CollectionUtils.merge(o1.properties, o2?.properties, false)
66 
67         o1
68     }
69 
70     void organization(Action<? super Organization> action) {
71         if (!organization) organization = new Organization()
72         action.execute(organization)
73     }
74 
75     void organization(@DelegatesTo(strategy = Closure.DELEGATE_FIRST, value = Organization) Closure<Void> action) {
76         if (!organization) organization = new Organization()
77         ConfigureUtil.configure(action, organization)
78     }
79 }
