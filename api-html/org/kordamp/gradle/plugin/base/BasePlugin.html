347     @Named('base')
348     private class BaseProjectEvaluatedListener implements ProjectEvaluatedListener {
349         @Override
350         void projectEvaluated(Project project) {
351             ProjectConfigurationExtension rootExtension = project.rootProject.extensions.findByType(ProjectConfigurationExtension)
352             ProjectConfigurationExtension extension = project.extensions.findByType(ProjectConfigurationExtension)
353             extension.normalize()
354 
355             boolean validate = checkFlag(ORG_KORDAMP_GRADLE_BASE_VALIDATE, true)
356 
357             List<String> errors = []
358             if (isRootProject(project)) {
359                 // extension == rootExtension
360                 extension.postMerge()
361             } else {
362                 // parent project may not have applied kordamp.base
363                 if (rootExtension) {
364                     extension.merge(rootExtension)
365                 } else {
366                     extension.postMerge()
367                 }
368             }
369 
370             if (validate) errors.addAll(extension.validate())
371 
372             if (validate && errors) {
373                 errors.each { project.logger.error(it) }
374                 boolean ideaSync = (System.getProperty('idea.sync.active') ?: 'false').toBoolean()
375                 if (!ideaSync) throw new GradleException("Project ${project.name} has not been properly configured.")
376             }
377         }
378     }

380     @Named('base')
381     private class BaseAllProjectsEvaluatedListener implements AllProjectsEvaluatedListener {
382         @Override
383         void allProjectsEvaluated(Project rootProject) {
384             // noop
385             if (checkFlag(ORG_KORDAMP_GRADLE_BASE_DEPENDENCY_MANAGEMENT, false)) {
386                 rootProject.allprojects(new Action<Project>() {
387                     @Override
388                     void execute(Project project) {
389                         resolveConfig(project).dependencyManagement.resolve()
390                     }
391                 })
392             }
393         }
