052     Publishing(ProjectConfigurationExtension config, Project project) {

056     @Override

061     @Override
062     Map<String, Map<String, Object>> toMap() {
063         Map<String, Object> map = new LinkedHashMap<String, Object>(enabled: enabled)
064 
065         map.signing = signing
066         map.releasesRepository = releasesRepository
067         map.snapshotsRepository = snapshotsRepository
068         map.publications = publications
069         map.scopes = scopes
070         map.useVersionExpressions = useVersionExpressions
071         map.flattenPlatforms = flattenPlatforms

073

075     }

077     void postMerge() {
078         if (!scopes) {
079             scopes << 'compile'
080             scopes << 'runtime'
081         }
082         if (isBlank(pom.packaging)) {
083             pom.packaging = 'jar'
084         }
085         super.postMerge()
086     }

123     static void merge(Publishing o1, Publishing o2) {
124         AbstractFeature.merge(o1, o2)
125         o1.releasesRepository = o1.@releasesRepository ?: o2.@releasesRepository
126         o1.snapshotsRepository = o1.@snapshotsRepository ?: o2.@snapshotsRepository
127         o1.@signing = o1.signingSet ? o1.signing : o2.signing
128         o1.@signingSet = o1.signingSet ?: o2.signingSet
129         o1.@useVersionExpressions = o1.useVersionExpressionsSet ? o1.useVersionExpressions : o2.useVersionExpressions
130         o1.@useVersionExpressionsSet = o1.useVersionExpressionsSet ?: o2.useVersionExpressionsSet
131         o1.@flattenPlatforms = o1.flattenPlatformsSet ? o1.flattenPlatforms : o2.flattenPlatforms
132         o1.@flattenPlatformsSet = o1.flattenPlatformsSet ?: o2.flattenPlatformsSet
133         o1.publications = CollectionUtils.merge(o1.publications, o2?.publications, false)
134         o1.scopes = CollectionUtils.merge(o1.scopes, o2?.scopes, false)
135         DefaultPomOptions.merge(o1.pom, o2.pom)
136     }
