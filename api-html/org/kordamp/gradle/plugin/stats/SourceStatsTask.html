051 @CacheableTask
052 class SourceStatsTask extends DefaultTask {
053     @Optional @Input Map<String, String> counters = [:]
054     @Optional @Input Map<String, Map<String, String>> paths = [:]
055     @Optional @Input List<String> formats = []
056 
057     @PathSensitive(PathSensitivity.RELATIVE)
058     @Optional @InputFile File reportDir
059 
060     @Internal
061     File xmlReport
062 
063     @Internal
064     int totalFiles = 0
065     @Internal
066     int totalLOC = 0
067 
068     @TaskAction
069     void computeLoc() {
070         if (!reportDir) reportDir = project.file("${project.reporting.baseDir.path}/stats")
071         if (!xmlReport) xmlReport = project.file("${reportDir}/${project.name}.xml")
072         reportDir.mkdirs()
073 
074         Map<String, Counter> counterInstances = resolveCounterInstances()
075 
076         ProjectConfigurationExtension config = resolveConfig(project)
077 
078         Map<String, Map<String, String>> merged = [:]
079         merged.putAll(config.stats.paths)
080         // deep copy
081         paths.each { key, val ->
082             Map<String, String> map = [:]
083             map.putAll(val)
084             merged.put(key, map)
085         }
086 
087         PluginUtils.resolveSourceDirs(project).each { File dir ->
088             if (!dir.exists()) return
089             dir.eachFileRecurse { File file ->
090                 if (file.file) {
091                     String extension = getFilenameExtension(file.name)
092                     Map map = merged.find { file.absolutePath =~ it.value.path && !it.value.extension }?.value
093                     if (!map) map = merged.find {
094                         file.absolutePath =~ it.value.path && extension == it.value.extension
095                     }?.value
096                     if (!map) map = merged.find { file.absolutePath =~ it.value.path }?.value
097                     if (!map || (map.extension && extension != map.extension)) return
098                     if (counterInstances.containsKey(extension)) {
099                         SourceStatsTask.countLines(map, counterInstances[extension], file)
100                     }
101                 }
102             }
103         }
104 
105         List<String> toBeRemoved = []
106         merged.each { type, info ->
107             if (info.files && info.lines) {
108                 totalFiles += info.files
109                 totalLOC += info.lines
110             } else {
111                 toBeRemoved << type
112             }
113         }
114         toBeRemoved.each { merged.remove(it) }
115 
116         if (totalFiles) {
117             int max = 0
118             merged.values().each { if (it.files) max = Math.max(max, it.name.size()) }
119             max = Math.max(max, 22)
120             merged = merged.sort { it.value.name }
121 
122             output(merged, max, totalFiles.toString(), totalLOC.toString(), new PrintWriter(System.out))
123             xmlOutput(merged, totalFiles.toString(), totalLOC.toString())
124             if (HTML in formats) htmlOutput(merged, totalFiles.toString(), totalLOC.toString())
125             if (TXT in formats) output(merged, max, totalFiles.toString(), totalLOC.toString(), new PrintWriter(getOutputFile(TXT)))
126         }
127     }
128 
129     private static void countLines(Map<String, Object> work, Counter counter, File file) {
130         int numFiles = work.get('files', 0)
131         work.files = ++numFiles
132         int lines = counter.count(file)
133         int numLines = work.get('lines', 0)
134         work.lines = numLines + lines
135     }

137     @CompileStatic
138     private Map<String, Counter> resolveCounterInstances() {
139         Map<String, Counter> instances = [:]
140         counters.collect { key, classname ->
141             instances[key] = (Counter) Class.forName(classname, true, SourceStatsTask.classLoader).newInstance()
142         }
143 
144         if (!instances.java) instances.java = new JavaCounter()
145         if (!instances.groovy) instances.groovy = new JavaCounter()
146         if (!instances.js) instances.js = new JavaCounter()
147         if (!instances.scala) instances.scala = new JavaCounter()
148         if (!instances.kt) instances.kt = new JavaCounter()
149         if (!instances.kts) instances.kt = new JavaCounter()
150         if (!instances.css) instances.css = new CssCounter()
151         if (!instances.scss) instances.scss = new JavaCounter()
152         if (!instances.xml) instances.xml = new XmlCounter()
153         if (!instances.html) instances.html = new XmlCounter()
154         if (!instances.fxml) instances.fxml = new XmlCounter()
155         if (!instances.sql) instances.sql = new SqlCounter()
156         if (!instances.yaml) instances.yaml = new HashCounter()
157         if (!instances.toml) instances.toml = new HashCounter()
158         if (!instances.clj) instances.clj = new SemiColonCounter()
159         if (!instances.get('properties')) instances.put('properties', new PropertiesCounter())
160 
161         instances
162     }

164     private void output(Map<String, Map<String, Object>> work, int max, String totalFiles, String totalLOC, Writer out) {
165         int padFiles = Math.max(totalFiles.toString().length(), 6)
166         int padLocs = Math.max(totalLOC.toString().length(), 6)
167 
168         out.println '    +-' + ('-' * max) + '-+--------+--------+'
169         out.println '    | ' + 'Name'.padRight(max, ' ') + ' | ' +
170             'Files'.padLeft(padFiles, ' ') + ' | ' +
171             'LOC'.padLeft(padLocs, ' ') + ' |'
172         out.println '    +-' + ('-' * max) + '-+--------+--------+'
173 
174         work.each { type, info ->
175             if (info.files) {
176                 out.println '    | ' +
177                     info.name.padRight(max, ' ') + ' | ' +
178                     info.files.toString().padLeft(padFiles, ' ') + ' | ' +
179                     info.lines.toString().padLeft(padLocs, ' ') + ' |'
180             }
181         }
182 
183         out.println '    +-' + ('-' * max) + '-+--------+--------+'
184         out.println '    | ' + 'Totals'.padRight(max, ' ') + ' | ' + totalFiles.padLeft(padFiles, ' ') + ' | ' + totalLOC.padLeft(padLocs, ' ') + ' |'
185         out.println '    +-' + ('-' * max) + '-+--------+--------+\n'
186 
187         out.flush()
188     }

190     private void xmlOutput(Map<String, Map<String, Object>> work, String totalFiles, String totalLOC) {
191         new MarkupBuilder(new FileWriter(getOutputFile(XML))).stats {
192             work.each { type, info ->
193                 if (info.files) {
194                     category(name: info.name) {
195                         fileCount(info.files.toString())
196                         loc(info.lines.toString())
197                     }
198                 }
199             }
200             category {
201                 name('Total')
202                 fileCount(totalFiles)
203                 loc(totalLOC)
204             }
205         }
206     }

208     private void htmlOutput(Map<String, Map<String, Object>> work, String totalFiles, String totalLOC) {
209         int i = 0
210         new MarkupBuilder(new FileWriter(getOutputFile(HTML))).html {
211             table(border: 1) {
212                 tr {
213                     th('Name')
214                     th(align: 'right', 'Files')
215                     th(align: 'right', 'LOC')
216                 }
217                 work.each { type, info ->
218                     if (info.files) {
219                         tr(style: (i++) % 2 ? 'background-color:lightblue' : 'background-color:FFF') {
220                             td(info.name)
221                             td(align: 'right', info.files.toString())
222                             td(align: 'right', info.lines.toString())
223                         }
224                     }
225                 }
226                 tr(style: 'background-color:lightgreen') {
227                     b {
228                         td('Total')
229                         td(align: 'right', totalFiles)
230                         td(align: 'right', totalLOC)
231                     }
232                 }
233             }
234         }
235     }
