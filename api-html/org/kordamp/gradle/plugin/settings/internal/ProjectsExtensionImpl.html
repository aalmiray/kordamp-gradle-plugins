069     ProjectsExtensionImpl(Settings settings, ObjectFactory objects) {
070         this.settings = settings
071         layout = objects.property(Layout).convention(Layout.TWO_LEVEL)
072         enforceNamingConvention = objects.property(Boolean).convention(true)
073         useLongPaths = objects.property(Boolean).convention(false)
074         cache = objects.property(Boolean).convention(true)
075         directories = objects.listProperty(String).convention([])
076         excludes = objects.setProperty(String).convention(new LinkedHashSet<String>())
077         prefix = objects.property(String).convention(Providers.notDefined())
078         suffix = objects.property(String).convention(Providers.notDefined())
079         fileNameTransformation = objects.property(FileNameTransformation).convention(Providers.notDefined())

081         settings.gradle.addBuildListener(new BuildAdapter() {
082             @Override
083             void settingsEvaluated(Settings s) {
084                 Cache.Key key = Cache.key(s.settingsDir.absolutePath + '-projects')
085                 if (!resolveFromCache(key)) {

088                 }

090             }

095             }

099     private boolean resolveFromCache(Cache.Key key) {
100         long settingsLastModified = findSettingsFile()?.lastModified() ?: Long.MIN_VALUE
101 
102         if (cache.get() &&
103             Cache.getInstance().has(settings.gradle, key) &&
104             Cache.getInstance().lastModified(settings.gradle, key) > settingsLastModified) {
105 
106             LOG.info("[settings] Reading project structure from cache. ${key.getAbsolutePath(settings.gradle)}")
107 
108             return Cache.getInstance().read(settings.gradle, key) { BufferedReader reader ->
109                 reader.readLines().each { String line ->
110                     String[] parts = line.split('#')
111                     String projectPath = parts[0]
112                     String projectName = parts[1]
113                     File projectDir = new File(parts[2])
114                     File buildFile = new File(parts[3])
115                     settings.include(projectPath)
116                     settings.project(projectPath).name = projectName
117                     settings.project(projectPath).projectDir = projectDir
118                     settings.project(projectPath).buildFileName = buildFile.name
119                     LOG.info("[settings] Including project ${projectPath} -> ${buildFile.absolutePath}")
120                 }
121             }
122         }
123 
124         false
125     }

127     private File findSettingsFile() {
128         [
129             settings.gradle.startParameter.settingsFile,
130             new File(settings.settingsDir, 'settings.gradle'),
131             new File(settings.settingsDir, 'settings.gradle.kts')
132         ].find { it?.exists() } ?: null
133     }

135     private void resolveFromConfig() {
136         LOG.info('[settings] Resolving project structure from configuration')
137 
138         switch (layout.get()) {
139             case Layout.TWO_LEVEL:
140                 processTwoLevelLayout()
141                 break
142             case Layout.MULTI_LEVEL:
143                 processMultiLevelLayout()
144                 break
145             case Layout.STANDARD:
146                 processStandardLayout()
147                 break
148             case Layout.EXPLICIT:
149                 break
150         }
151         // Always process explicit
152         processExplicitDirectoriesAndPaths()
153         cleanup()
154     }

156     private void writeConfigToCache(Cache.Key key) {
157         LOG.info("[settings] Writing project structure to cache. ${key.getAbsolutePath(settings.gradle)}")
158 
159         Cache.getInstance().write(settings.gradle, key) { BufferedWriter writer ->
160             writeProjectToCache(settings.rootProject, writer)
161         }
162     }

164     private void writeProjectToCache(ProjectDescriptor project, BufferedWriter writer) {
165         writer.writeLine([
166             project.path,
167             project.name,
168             project.projectDir.absolutePath,
169             project.buildFile.absolutePath
170         ].join('#'))
171         for (ProjectDescriptor child : project.children) {
172             writeProjectToCache(child, writer)
173         }
174     }

228     private void processExplicitDirectoriesAndPaths() {
229         for (DirectorySpecImpl spec : directoryConditions) {
230             if (!spec.applies()) continue
231             File parentDir = new File(settings.rootDir, spec.dir)
232             if (!parentDir.exists()) {
233                 LOG.info "Skipping ${parentDir} as it does not exist."
234                 continue
235             }
236 
237             excludes.addAll(spec.excludes)
238             doProcessTwoLevelLayout(parentDir)
239         }
240 
241         for (PathSpecImpl spec : pathConditions) {
242             if (!spec.applies()) continue
243             File projectDir = new File(settings.rootDir, spec.path)
244             if (!projectDir.exists()) {
245                 LOG.info "Skipping ${projectDir} as it does not exist."
246                 continue
247             }
248             doIncludeProject(spec.path)
249         }
250     }
251 
252     private void processTwoLevelLayout() {
253         if (directories.get()) {
254             for (String parentDirName : directories.get()) {
255                 File parentDir = new File(settings.rootDir, parentDirName)
256                 if (!parentDir.exists()) {
257                     LOG.info "Skipping ${parentDir} as it does not exist."
258                     continue
259                 }
260 
261                 doProcessTwoLevelLayout(parentDir)
262             }
263         } else {
264             settings.settingsDir.eachDir { File parentDir ->
265                 if (!skipDirectoryDiscoveryFor(parentDir)) {
266                     doProcessTwoLevelLayout(parentDir)
267                 }
268             }
269         }
270     }
271 
272     private void processMultiLevelLayout() {
273         if (directories.get()) {
274             for (String path : directories.get()) {
275                 File projectDir = new File(settings.rootDir, path)
276                 if (!projectDir.exists()) {
277                     LOG.info "Skipping ${projectDir} as it does not exist."
278                     continue
279                 }
280                 doIncludeProject(path)
281             }
282         } else {
283             settings.settingsDir.eachDir { File projectDir ->
284                 discoverProject(projectDir)
285             }
286         }
287     }
288 
289     private void discoverProject(File projectDir) {
290         if (!skipDirectoryDiscoveryFor(projectDir)) {
291             // Is there a matching build file?
292             String projectDirName = projectDir.name
293             File buildFile = resolveBuildFile(projectDir, projectDirName)
294             // yes -> include and stop
295             if (buildFile && buildFile.exists()) {
296                 String projectPath = resolveProjectPath(projectDir, projectDirName)
297 
298                 settings.include(projectPath)
299                 settings.project(projectPath).name = projectDirName
300                 settings.project(projectPath).projectDir = projectDir
301                 settings.project(projectPath).buildFileName = buildFile.name
302                 LOG.info("[settings] Including project ${projectPath} -> ${buildFile.absolutePath}")
303             } else {
304                 // no -> continue search
305                 projectDir.eachDir { File dir ->
306                     discoverProject(dir)
307                 }
308             }
309         }
310     }

312     private void processStandardLayout() {
313         settings.rootDir.eachDir { File projectDir ->
314             if (isProjectExcluded(projectDir.name, projectDir.name)) return
315 
316             File buildFile = resolveBuildFile(projectDir, projectDir.name)
317             doIncludeProject(settings.rootDir, projectDir.name, buildFile)
318         }
319     }

321     private boolean skipDirectoryDiscoveryFor(File dir) {
322         dir.name.charAt(0) == '.' ||
323             (dir.name in ['buildSrc', 'gradle'] && settings.rootProject.projectDir == dir.parentFile) ||
324             (dir.name == 'build' && settings.findProject(dir.parentFile))
325     }
326 
327     private void doProcessTwoLevelLayout(File parentDir) {
328         parentDir.eachDir { File projectDir ->
329             if (isProjectExcluded("${projectDir.parentFile.name}/${projectDir.name}", projectDir.name)) return
330 
331             File buildFile = resolveBuildFile(projectDir, projectDir.name)
332             doIncludeProject(parentDir, projectDir.name, buildFile)
333         }
334     }

336     private void doIncludeProject(File parentDir, String projectDirName) {
337         File buildFile = resolveBuildFile(parentDir, projectDirName)
338         doIncludeProject(settings.rootDir, parentDir.name, buildFile)
339     }

341     private void doIncludeProject(File projectDir) {
342         doIncludeProject(projectDir.parentFile, projectDir.name)
343     }

345     private void doIncludeProject(String path) {
346         String[] parts = path.split(Pattern.quote(File.separator))
347         String projectDirName = path
348         String projectName = parts[-1]
349 
350         if (isProjectExcluded(projectDirName, projectName)) return
351 
352         File projectDir = new File(settings.rootDir, projectDirName)
353 
354         assert projectDir.isDirectory()
355 
356         File buildFile = resolveBuildFile(projectDir, projectName)
357         if (buildFile && buildFile.exists()) {
358             String projectPath = resolveProjectPath(projectDir, projectName)
359 
360             settings.include(projectPath)
361             settings.project(projectPath).name = projectName
362             settings.project(projectPath).projectDir = projectDir
363             settings.project(projectPath).buildFileName = buildFile.name
364             LOG.info("[settings] Including project ${projectPath} -> ${buildFile.absolutePath}")
365         }
366     }

368     private boolean isProjectExcluded(String projectDir, String projectName) {
369         if (excludes.get().contains(projectName)) return true
370 
371         for (String exclude : excludes.get()) {
372             if (pattern(exclude).matcher(projectDir).matches()) return true
373         }
374         for (String exclude : excludes.get()) {
375             if (pattern(exclude).matcher(projectName).matches()) return true
376         }
377 
378         false
379     }

381     @Memoized
382     private Pattern pattern(String regex) {
383         Pattern.compile(asGlobRegex(regex))
384     }

386     private File resolveBuildFile(File projectDir, String projectName) {
387         if (enforceNamingConvention.get()) {
388             if (FileNameTransformation.ADD == fileNameTransformation.orNull) {
389                 if (!isBlank(prefix.orNull)) {
390                     projectName = prefix.get() + projectName
391                 }
392                 if (!isBlank(suffix.orNull)) {
393                     projectName += suffix.get()
394                 }
395             } else if (FileNameTransformation.REMOVE == fileNameTransformation.orNull) {
396                 if (!isBlank(prefix.get())) {
397                     projectName -= prefix.get()
398                 }
399                 if (!isBlank(suffix.get())) {
400                     projectName -= suffix.get()
401                 }
402             }
403         }
404 
405         for (String fileName : [
406             projectName + '.gradle'.toString(),
407             'build.gradle',
408             projectName + '.gradle.kts'.toString(),
409             'build.gradle.kts']) {
410             File buildFile = new File(projectDir, fileName)
411             if (buildFile.exists()) {
412                 return buildFile
413             }
414 
415         }
416         null
417     }
418 
419     private void doIncludeProject(File parentDir, String projectDirName, File buildFile) {
420         if (buildFile && buildFile.exists()) {
421             File projectDir = new File(parentDir, projectDirName)
422             String projectPath = resolveProjectPath(projectDir, projectDirName)
423 
424             settings.include(projectPath)
425             settings.project(projectPath).name = projectDirName
426             settings.project(projectPath).projectDir = projectDir
427             settings.project(projectPath).buildFileName = buildFile.name
428             LOG.info("[settings] Including project ${projectPath} -> ${buildFile.absolutePath}")
429         }
430     }

432     private String resolveProjectPath(File projectDir, String projectName) {
433         String projectPath = ':' + projectName
434         if (getUseLongPaths().get()) {
435             projectPath = projectDir.absolutePath - settings.rootDir.absolutePath
436             if (projectPath.endsWith(File.separator)) projectPath = projectPath[0..-2]
437             projectPath = projectPath.replace(File.separator, ':')
438         }
439         projectPath
440     }

442     private void cleanup() {
443         ProjectDescriptor pd = settings.rootProject
444         for (ProjectDescriptor child : pd.children) {
445             purgeEmptyProjects(child, pd)
446         }
447     }
448 
449     private void purgeEmptyProjects(ProjectDescriptor project, ProjectDescriptor parent) {
450         if (!project.children) {
451             if (!project.buildFile.exists()) {
452                 parent.children.remove(project)
453             }
454         } else {
455             for (ProjectDescriptor child : project.children) {
456                 purgeEmptyProjects(child, project)
457             }
458 
459             if (!project.buildFile.exists()) {
460                 parent.children.remove(project)
461                 parent.children.addAll(project.children)
462             }
463         }
464     }

466     private boolean isBlank(String str) {
467         if (str == null || str.length() == 0) {
468             return true
469         }
470         for (char c : str.toCharArray()) {
471             if (!Character.isWhitespace(c)) {
472                 return false
473             }
474         }
475 
476         return true
477     }
