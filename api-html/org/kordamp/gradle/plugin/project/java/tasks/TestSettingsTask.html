018 package org.kordamp.gradle.plugin.project.java.tasks
019 
020 import groovy.transform.CompileStatic
021 import org.gradle.api.tasks.TaskAction
022 import org.gradle.api.tasks.testing.Test
023 import org.gradle.api.tasks.testing.junit.JUnitOptions
024 import org.gradle.api.tasks.testing.junitplatform.JUnitPlatformOptions
025 import org.gradle.api.tasks.testing.testng.TestNGOptions
026 import org.kordamp.gradle.plugin.base.tasks.AbstractSettingsTask
027 
028 /**
029  * @author Andres Almiray
030  * @since 0.30.0
031  */
032 @CompileStatic
033 class TestSettingsTask extends AbstractSettingsTask {
034     @TaskAction
035     void report() {
036         if (tasks) {
037             tasks.each { t ->
038                 printTask((Test) project.tasks.findByName(t))
039             }
040         } else if (task) {
041             try {
042                 printTask((Test) project.tasks.findByName(task))
043             } catch (NullPointerException e) {
044                 throw new IllegalStateException("No matching '${this.task}' task was found")
045             }
046         } else {
047             Set<Test> testTasks = new LinkedHashSet<>(project.tasks.withType(Test))
048             testTasks.each { t ->
049                 printTask(t)
050             }
051         }
052     }
053 
054     private void printTask(Test task) {
055         print(task.name + ':', 0)
056         doPrintCollection('includes', task.includes, 1)
057         doPrintCollection('excludes', task.excludes, 1)
058         doPrintMapEntry('debug', task.debug, 1)
059         doPrintMapEntry('enableAssertions', task.enableAssertions, 1)
060         doPrintMapEntry('executable', task.executable, 1)
061         doPrintMapEntry('failFast', task.failFast, 1)
062         doPrintMapEntry('forkEvery', task.forkEvery, 1)
063         doPrintMapEntry('ignoreFailures', task.ignoreFailures, 1)
064         doPrintCollection('jvmArgs', task.jvmArgs, 1)
065         doPrintMapEntry('minHeapSize', task.minHeapSize, 1)
066         doPrintMapEntry('maxHeapSize', task.maxHeapSize, 1)
067         doPrintMapEntry('maxParallelForks', task.maxParallelForks, 1)
068         doPrintMapEntry('scanForTestClasses', task.scanForTestClasses, 1)
069         doPrintMap('systemProperties', task.systemProperties, 1)
070         print('options:', 1)
071         if (task.options instanceof JUnitOptions) {
072             JUnitOptions o = (JUnitOptions) task.options
073             doPrintMapEntry('useJUnit', true, 2)
074             doPrintCollection('includeCategories', o.includeCategories, 2)
075             doPrintCollection('excludeCategories', o.excludeCategories, 2)
076         } else if (task.options instanceof JUnitPlatformOptions) {
077             JUnitPlatformOptions o = (JUnitPlatformOptions) task.options
078             doPrintMapEntry('useJUnitPlatform', true, 2)
079             doPrintCollection('includeEngines', o.includeEngines, 2)
080             doPrintCollection('includeTags', o.includeTags, 2)
081             doPrintCollection('excludeEngines', o.excludeEngines, 2)
082             doPrintCollection('excludeTags', o.excludeTags, 2)
083         } else if (task.options instanceof TestNGOptions) {
084             TestNGOptions o = (TestNGOptions) task.options
085             doPrintMapEntry('useTestNG', true, 2)
086             doPrintMapEntry('configFailurePolicy', o.configFailurePolicy, 2)
087             doPrintCollection('includeGroups', o.includeGroups, 2)
088             doPrintCollection('excludeGroups', o.excludeGroups, 2)
089             doPrintMapEntry('groupByInstances', o.groupByInstances, 2)
090             doPrintCollection('listeners', o.listeners, 2)
091             doPrintMapEntry('parallel', o.parallel, 2)
092             doPrintMapEntry('preserveOrder', o.preserveOrder, 2)
093             doPrintMapEntry('suiteName', o.suiteName, 2)
094             doPrintMapEntry('threadCount', o.threadCount, 2)
095             doPrintMapEntry('useDefaultListeners', o.useDefaultListeners, 2)
096         }
097         if (isShowPaths()) {
098             doPrintCollection('bootstrapClasspath', task.bootstrapClasspath, 1)
099             doPrintCollection('classpath', task.classpath, 1)
100         }
101         println ' '
102     }
103 }
