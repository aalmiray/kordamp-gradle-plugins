018 package org.kordamp.gradle.plugin.publishing
019 
020 import groovy.transform.CompileStatic
021 import org.gradle.api.publish.Publication
022 import org.gradle.api.publish.PublishingExtension
023 import org.gradle.api.publish.maven.MavenPublication
024 import org.gradle.api.tasks.TaskAction
025 import org.gradle.api.tasks.options.Option
026 import org.kordamp.gradle.plugin.base.tasks.AbstractReportingTask
027 
028 import static org.kordamp.gradle.util.StringUtils.isNotBlank
029 
030 /**
031  * @author Andres Almiray
032  * @since 0.20.0
033  */
034 @CompileStatic
035 class PublicationSettingsTask extends AbstractReportingTask {
036     private boolean absolute
037     private String publication
038     private Set<String> publications
039 
040     @Option(option = 'absolute', description = "Should paths be printed as absolutes or not. Defaults to 'false'")
041     void setAbsolute(boolean absolute) {
042         this.absolute = absolute
043     }
044 
045     @Option(option = 'publication', description = 'The publication to generate the report for.')
046     void setPublication(String publication) {
047         this.publication = publication
048     }
049 
050     @Option(option = 'publications', description = 'The publications to generate the report for.')
051     void setPublications(String publications) {
052         if (isNotBlank(publications)) {
053             this.publications = (publications.split(',').collect { it.trim() }) as Set
054         }
055     }
056 
057     @TaskAction
058     void report() {
059         PublishingExtension publishing = project.extensions.findByType(PublishingExtension)
060         if (publishing) {
061             if (publications) {
062                 publications.each { p ->
063                     printPublication(publishing.publications.findByName(p))
064                 }
065             } else if (publications) {
066                 printPublication(publishing.publications.findByName(publication))
067             } else {
068                 publishing.publications.each { Publication p ->
069                     printPublication(p)
070                 }
071             }
072         }
073     }
074 
075     private void printPublication(Publication publication) {
076         print(publication.name + ':', 0)
077         if (publication instanceof MavenPublication) {
078             doPrintMapEntry('type', 'Maven', 1)
079             MavenPublication p = (MavenPublication) publication
080             doPrintMapEntry('groupId', p.groupId, 1)
081             doPrintMapEntry('artifactId', p.artifactId, 1)
082             doPrintMapEntry('version', p.version, 1)
083             if (p.artifacts.size()) {
084                 print('artifacts:', 1)
085                 int index = 0
086                 p.artifacts.each { artifact ->
087                     print('artifact ' + (index++) + ':', 2)
088                     doPrintMapEntry('file', adjustPath(project.rootProject.projectDir.toString(), artifact.file?.absolutePath), 3)
089                     doPrintMapEntry('classifier', artifact.classifier, 3)
090                     doPrintMapEntry('extension', artifact.extension, 3)
091                 }
092             }
093         }
094         println ' '
095     }
096 
097     private String adjustPath(String rootDir, String path) {
098         if (isNotBlank(path) && !absolute) {
099             return (path - rootDir)[1..-1]
100         }
101         return path
102     }
103 }
